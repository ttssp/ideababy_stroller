### `pars/cli/fleet.py`(本 task 仅 status 部分)
```python
import typer
fleet_app = typer.Typer(help="Fleet management commands (v0.2)")

@fleet_app.command("status")
def fleet_status(
    runs_root: Path = typer.Option(..., envvar="RECALLKIT_RUNS_DIR"),
    show_all: bool = typer.Option(False, "--all", help="include completed/failed"),
    fleet_name: str | None = typer.Option(None, "--fleet", help="filter by fleet name"),
):
    rows = list_active_runs(runs_root) if not show_all else list_all_runs(runs_root)
    if fleet_name:
        rows = [r for r in rows if r.fleet_name == fleet_name]
    locks = list_gpu_locks(runs_root / ".gpu_locks")
    typer.echo(render_status_table(rows, locks))
```
