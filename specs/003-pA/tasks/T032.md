### `pars/cli/fleet.py`(本 task 加 compare 子命令)
```python
@fleet_app.command("compare")
def fleet_compare(
    fleet_name: str = typer.Argument(...),
    eval_metric: str = typer.Option("gsm8k", "--metric"),
    runs_root: Path = typer.Option(..., envvar="RECALLKIT_RUNS_DIR"),
):
    result = compare_fleet(fleet_name, runs_root, eval_metric=eval_metric)
    typer.echo(result.markdown)
```
