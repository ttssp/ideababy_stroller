  ```python
  class BLiteService:
      """B-lite 降级路径 (R3: CLI only, 显式调 pause hook)."""

      def engage(self, reason: str) -> MetaDecision:
          # 显式调用 (不再"假设接口存在")
          from decision_ledger.monitor.pause_pipeline import pause_all_pipelines
          pause_all_pipelines()  # T010.pause + T015.pause
          # 写 meta_decisions
          meta = MetaDecision(
              decision_type='b_lite_engaged',
              cooling_off_until=datetime.now() + timedelta(days=14),
              reason=reason,
          )
          self.repo.insert(meta)
          return meta

      def disengage(self) -> MetaDecision:
          last = self.repo.last_engage()
          if last and datetime.now() < last.cooling_off_until:
              raise ValueError(f"cooling-off until {last.cooling_off_until}, cannot disengage")
          from decision_ledger.monitor.pause_pipeline import unpause_all_pipelines
          unpause_all_pipelines()
          meta = MetaDecision(decision_type='b_lite_disengaged', ...)
          self.repo.insert(meta)
          return meta

      def is_engaged(self) -> bool:
          # 读 T010 + T015 hook 实际状态
          from decision_ledger.services.conflict_worker import conflict_worker
          return conflict_worker.is_paused()

      def status(self) -> dict:
          """CLI status 输出: engaged + cooling_off_remaining + last_reason."""
          ...
  ```
