  ```python
  import os
  import logging
  logger = logging.getLogger(__name__)

  def pause_all_pipelines():
      """B-lite 一键降级 (R3: ImportError 不静默吞)."""
      conflict_worker.pause()  # T010 hook (强依赖, 必 import)
      try:
          from decision_ledger.services.monthly_scheduler import monthly_scheduler
          monthly_scheduler.pause()  # T015 hook
      except ImportError as e:
          # R3 ⚠️ H1: T015 未 ship 是 graceful no-op (warning), T015 ship 后 import 失败 = wiring 错误
          logger.warning(f"MonthlyScheduler import failed (T015 not shipped or wiring broken): {e}")
          if os.environ.get("DECISION_LEDGER_TEST_MODE") == "strict":
              # CI / test mode 下视为失败, 防 wiring drift
              raise AssertionError(f"T015 import expected to succeed in strict test mode: {e}")
      logger.info("All pipelines paused (B-lite engaged)")

  def unpause_all_pipelines():
      conflict_worker.resume()
      try:
          from decision_ledger.services.monthly_scheduler import monthly_scheduler
          monthly_scheduler.resume()
      except ImportError as e:
          logger.warning(f"MonthlyScheduler import failed: {e}")
          if os.environ.get("DECISION_LEDGER_TEST_MODE") == "strict":
              raise AssertionError(f"T015 import expected to succeed: {e}")
  ```
